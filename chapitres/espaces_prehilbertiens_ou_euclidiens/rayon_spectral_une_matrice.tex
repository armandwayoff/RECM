\begin{defi}{Rayon spectral}
    Soient $n \geqslant 2, M \in \M_n(\C)$. On définit son \emph{rayon spectral}:
    $$\rho(M) \defeq \max \{ |\lambda |,\ \lambda \in \Sp_{\C}(M) \}.$$
\end{defi}
