\begin{tikzpicture}

  % https://copyprogramming.com/howto/how-do-i-draw-arrows-at-coordinate-on-a-plot
  
  \def\xmax{4.5}
  \def\ymax{1.3}
  \def\ymin{-1.5}
  \def\xzero{2}
  \def\x{5}

  \pgfmathdeclarefunction{f}{1}{%
    \pgfmathparse{2*exp(-#1^2/2)-(#1/7)^2-1}%
  }
  
  \def\colfonc{mygreen}

    \begin{axis}[
        restrict x to domain=0:8,
        samples=100,
        ticks=none,
        xmin = -1, xmax = \xmax+0.5,
        ymin = \ymin, ymax = \ymax,
        unbounded coords=jump,
        axis x line=middle,
        axis y line=middle,
        axis line style={-latex, thick},
        ]           
      \draw[dashed] (axis cs:0,{f(\xzero)}) node[left=1pt] (l) {$f(x_0)$} -| (axis cs:2,0) node[above] (a) {$x_0$}; 
      \draw (axis cs:\x-0.5,0) node[above] {$x$};

      \addplot[fill=mylightblue, draw=none, domain=\xzero:\xmax] {f(x)} \closedcycle;

      \fill[pattern color=black, pattern=north east lines] (\xzero,0) rectangle (\x-0.5,f(\xzero);
    
      \contourlength{2pt}
      \node at ({(\xzero+\xmax)/2},{f(\xzero)/2}) {\contour{mylightblue}{$f(x_0) (x - x_0)$}};

      \addplot[name path=A,very thick,color=\colfonc, mark=none, domain=0:\xmax] {f(x)} node [above=10pt,near start] {$f$};

      \draw[->, black, thick] (\xzero,{1.5*f(\xzero)}) node[left] {\color{myblue}$\displaystyle \int_{x_0}^x f(t)\, \mathrm{d} t$} to [out=0,in=-110] (\xzero/2+\xmax/2,{1.2*f(\xzero)});  

    \end{axis}
\end{tikzpicture}
