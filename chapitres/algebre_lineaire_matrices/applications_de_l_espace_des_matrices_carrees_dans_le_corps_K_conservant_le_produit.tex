\begin{defi}{Forme multiplicative}
    Soit $f$ une application de $\M_n(\K)$ dans $\K$. L'application $f$ est dite \emph{multiplicative} si pour tout matrices $A$ et $B$ de $\M_n(\K)$, $f(AB) = f(A)f(B)$.
\end{defi}
