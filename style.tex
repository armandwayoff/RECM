% Format titre partie
\makeatletter
\renewcommand*\l@part[2]{%
  \ifnum \c@tocdepth >-2\relax
    \addpenalty{-\@highpenalty}%
    \addvspace{2.25em \@plus\p@}%
    \begingroup
      \parindent \z@ 
      \rightskip \z@
      \parfillskip \z@
      \leavevmode
      {\Large\bfseries
       \centering \MakeUppercase{#1}\par}% Actually center the text
      \nobreak
      \global\@nobreaktrue
      \everypar{\global\@nobreakfalse\everypar{}}%
    \endgroup
  \fi}
\makeatother
